DECLARE
 VAR1 VARCHAR2(40);
-- WNM_VDD VENDEDOR.NM_VDD%TYPE;
-- WVL_SAL NUMBER(9,2);
 WREG_VDD VENDEDOR%ROWTYPE;
 BEGIN
  VAR1 := 'LUCIANO';
  DBMS_OUTPUT.PUT_LINE('OLÁ '||VAR1);
  BEGIN
    SELECT * INTO WREG_VDD
    FROM VENDEDOR WHERE CD_VDD=1;
    IF WREG_VDD.VL_SAL < 1000 THEN
      DBMS_OUTPUT.PUT_LINE('VENDEDOR= '||WREG_VDD.NM_VDD||
      ' SALARIO= BAIXO');
    ELSIF WREG_VDD.VL_SAL >=1000 AND WREG_VDD.VL_SAL < 2500 THEN
      DBMS_OUTPUT.PUT_LINE('VENDEDOR= '||WREG_VDD.NM_VDD||
      ' SALARIO= RAZOÁVEL');
    ELSE
      DBMS_OUTPUT.PUT_LINE('VENDEDOR= '||WREG_VDD.NM_VDD||
      ' SALARIO= ALTO');
    END IF;
    
  EXCEPTION
    WHEN TOO_MANY_ROWS THEN
      DBMS_OUTPUT.PUT_LINE('RETORNADOS MUITOS VENDEDORES!!');
  END;
  DECLARE
    WNM_CLI VARCHAR2(40);
  BEGIN
    SELECT NM_CLI INTO WNM_CLI
    FROM CLIENTE WHERE CD_CLI>2;
    DBMS_OUTPUT.PUT_LINE('CLIENTE= '||WNM_CLI);
  EXCEPTION
   WHEN NO_DATA_FOUND THEN
    DBMS_OUTPUT.PUT_LINE('CLIENTE INEXISTENTE!!!!');
  END;
 EXCEPTION
  WHEN OTHERS THEN
   DBMS_OUTPUT.PUT_LINE('ERRO NÃO ESPERADO!! CONTATE O ADMINISTRADOR!!');
 END;
